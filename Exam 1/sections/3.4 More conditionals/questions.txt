[3.4.8] Multiple Choice: In the following switch statement, what will be the value of `coin`?
let change = 10; 
switch (change) { 
    case 1: coin = "penny"; 
        break; 
    case 5: coin = "nickel"; 
        break; 
    case 10: coin = "dime"; 
        break;
    case 25: coin = "quarter"; 
        break; 
    default: coin = "unknown"; } 
    console.log(coin);

- (A) "penny"
- (B) "nickel"
- (C) "dime"
- (D) "unknown"
